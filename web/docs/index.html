Quick API Example

JavaScript - Verify Product Authenticity 
// Initialize TAG IT SDK
import { TagIt } from '@tagit/sdk';

const tagit = new TagIt({
  apiKey: process.env.TAGIT_API_KEY,
  network: 'mainnet'
});

// Verify product authenticity by NFC tag ID
async function verifyProduct(tagId) {
  const result = await tagit.products.verify(tagId);

  return {
    authentic: result.verified,
    product: result.metadata,
    owner: result.currentOwner,
    history: result.transferHistory
  };
}
